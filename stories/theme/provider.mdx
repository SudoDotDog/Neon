# Neon Theme Provider


- Without provider

<div>
    {this.props.example(this.props.label)}
</div>

- With provider

<NeonThemeProvider value={{
    margin: MARGIN.SMALL,
    main: {
        color: 'red',
    },
}}>
    {this.props.example(this.props.label)}
</NeonThemeProvider>
